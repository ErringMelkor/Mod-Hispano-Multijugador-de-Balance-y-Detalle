 age_of_nicholas_i.2 = { #Nicholas I brings Kiselyov as IG Leader.
	type = country_event
	placement = root

	title = age_of_nicholas_i.2.t
	desc = age_of_nicholas_i.2.d
	flavor = age_of_nicholas_i.2.f

	left_icon = scope:tsar_scope

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/law_icons/monarchy.dds"

	duration = 3

	cooldown = { days = stupidly_long_modifier_time }

	trigger = {
		#Fired scripted_button.
	}

	immediate = {
		ruler = {
			save_scope_as = tsar_scope
		}
		create_character = {
			template = RUS_Pavel_Kiselyov
			on_created = {
				set_variable = kiselyov_var
			}
		}
		random_scope_character = {
			limit = {
				has_variable = kiselyov_var
			}
			save_scope_as = kiselyov_scope
		}
	}

	option = {
		name = age_of_nicholas_i.2.a
		default_option = yes
		show_as_tooltip = {
			create_character = {
				template = RUS_Pavel_Kiselyov
			}
		}

	}

	option = {
		name = age_of_nicholas_i.2.b
		show_as_tooltip = {
			create_character = {
				template = RUS_Pavel_Kiselyov
			}
		}
		scope:kiselyov_scope = {
			add_modifier = {
				name = modifier_man_of_the_people
			}
			add_trait = imperious
			add_trait = reckless
		}
	}
}
