bourbon_instability_variables = {
	#bourbon Variables
	set_variable = { name = bourbon_progress_from_legitimacy value = 0 }

	set_variable = { name = ig_weight value = 0 } #This variable is used solely for mathematical purposes, and is set as equivalent to the clout of a given interest group. It exists so that IGs can be weighted higher or lower depending on whether they're in-government or not.
}

bourbon_instability_pulse_variables_progress = {
	set_variable = { name = bourbon_progress_from_legitimacy value = root.government_legitimacy }
	# From Legitimacy
	if = {
		limit = {
			government_legitimacy <= 50
		}
		change_variable = { name = bourbon_progress_from_legitimacy add = 1 }
	}
	else_if = {
		limit = {
			government_legitimacy > 50
		}
		change_variable = { name = bourbon_progress_from_legitimacy subtract = 0.5 }
	}
	else_if = {
		limit = {
			government_legitimacy >= 25
		}
		change_variable = { name = bourbon_progress_from_legitimacy subtract = 2 }
	}
}
