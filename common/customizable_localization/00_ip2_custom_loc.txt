get_king_sweden_norway = {
	type = country
	text = {
		trigger = {
			AND = {
				OR = {
					any_subject_or_below = {
						c:NOR ?= this
					}
					any_scope_state = {
						state_region = s:STATE_EASTERN_NORWAY
					}
				}
				OR = {
					any_subject_or_below = {
						c:DEN ?= this
					}
					any_scope_state = {
						state_region = s:STATE_ZEALAND
					}
				}
			}
		}
		localization_key = king_of_sweden_denmark_and_norway_key
	}
