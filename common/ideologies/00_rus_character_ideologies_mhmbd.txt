ideology_nicholean_reformer_leader = {
	icon = "gfx/interface/icons/ideology_icons/ideology_leader/unused/ideology_leader_paternalistic.dds"

	character_ideology = yes

	lawgroup_bureaucracy = {
		law_appointed_bureaucrats = approve
		law_hereditary_bureaucrats = neutral
		law_elected_bureaucrats = disapprove
	}

	lawgroup_education_system = {
		law_public_schools = approve
		law_private_schools = approve
		law_religious_schools = neutral
		law_no_schools = disapprove
	}

	lawgroup_land_reform = {
		law_serfdom = disapprove
		law_tenant_farmers = approve
		law_collectivized_agriculture = strongly_disapprove
		law_commercialized_agriculture = disapprove
		law_peasant_proprietorship = strongly_disapprove
	}

	lawgroup_childrens_rights = {
		law_child_labor_allowed = disapprove
		law_restricted_child_labor = neutral
		law_compulsory_primary_school = approve
	}

	interest_group_leader_trigger = {
		ideology_nicholean_reformer_valid_trigger = yes
	}

	country_trigger = {
		c:RUS ?= this

		game_date < 1860.1.1
	}

	non_interest_group_leader_trigger = {
		# Agitators should have an ideology at odds with current laws
		NAND = {
			has_role = agitator
			owner = {
				has_law_or_variant = law_type:law_appointed_bureaucrats
				has_law_or_variant = law_type:law_compulsory_primary_school
			}
		}
	}

	interest_group_leader_weight = {
		value = 75 #poco probable


		# less likely among more progressive IG's if they're angry and socialism is researched
		if = {
			limit = {
				owner = {
					has_technology_researched = socialism
				}
				OR = {
					is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_trade_unions
				}
				ig_approval < 0
			}
			add = {
				value = ig_approval
				multiply = 5
			}
		}

		# más probable temporalmente si se ha completado la entrada con éxito
		if = {
			limit = {
				has_variable = nicholas_completed_var
			}
			add = 150
		}

		#más probable si tienes burócratas a dedo
		if = {
			limit = {
				has_law = law_type:law_appointed_bureaucrats
			}
			add = 50
		}

		#más probable si tienes escuelas públicas o gimnasios.
		if = {
			limit = {
				OR = {
					has_law = law_type:law_gymnazii
					has_law = law_type:law_public_schools
				}
			}
			add = 50
		}

		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_ethno_nationalism
					has_variable = chose_conservatism
					has_variable = chose_fascism
				}
			}
			multiply = 0.1
		}
	}
}

	non_interest_group_leader_trigger = {
		NAND = {
			has_role = agitator
			owner = {
				has_law_or_variant = law_type:law_monarchy
				has_law_or_variant = law_type:law_national_supremacy
				has_law_or_variant = law_type:law_right_of_assembly

		interest_group = {
			ideology_zapadniki_valid_trigger = yes

		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_monarchy
				}
			}
			add = 0
		}
		else = {
			subtract = 500
		}

		# improbable, pero posible, en landowners.
		if = {
			limit = {
				is_interest_group_type = ig_landowners
			}
			add = -50
		}

		# More likely if issues are relevant - greater bonus to this than usual
		if = {
			limit = {
				owner.ideology_zapadniki_disfavoured_laws > 0
			}
			add = {
				value = owner.ideology_zapadniki_disfavoured_laws
				multiply = 32
			}
		}

		# Less likely if issues are not relevant
		if = {
			limit = {
				owner.ideology_zapadniki_favoured_laws > 0
			}
			subtract = {
				value = owner.ideology_zapadniki_favoured_laws
				multiply = 8
			}
		}
	}
}
