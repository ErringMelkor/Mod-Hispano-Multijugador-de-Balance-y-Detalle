je_age_of_alexander_ii = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_historical_content

	is_shown_in_lobby = {
		c:RUS ?= this
	}

	immediate = {
		random_scope_character = {
			limit = {
				has_template = RUS_alexander_ii
			}
			save_scope_as = RUS_alexander_ii_scope
		}
	}

	complete = {
		trigger_if = {
			limit = {
				exists = scope:RUS_alexander_ii_scope
			}
			scope:RUS_alexander_ii_scope ?= {
				is_ruler = no
			}
			# se muere de manera natural.
		}
	}

	on_complete = {
		remove_modifier = modifier_alexander_ii_rule
		add_modifier = {
			name = modifier_alexander_ii_legacy
			years = 10
			multiplier = alexander_ii_agenda_value
		}
	}

	on_monthly_pulse = {
		random_events = {
		}
		effect = {

			remove_modifier = modifier_alexander_ii_rule
			add_modifier = {
				name = modifier_alexander_ii_rule
				multiplier = alexander_ii_agenda_value
			}
		}
	}

	fail = {
	}

	on_fail = {
		remove_modifier = modifier_alexander_ii_rule
		add_modifier = {
			name = modifier_alexander_ii_disaster
			years = 10
		}
	}

	invalid = {
		NOT = {
			has_law_or_variant = law_type:law_monarchy
		}
	}

	weight = 1000
	should_be_pinned_by_default = yes
}
