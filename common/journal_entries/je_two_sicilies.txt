	is_shown_in_lobby = {
		c:SIC ?= THIS
	}
	
	group = je_group_historical_content

	modifiers_while_active = {
		bourbon_instability
	}

	immediate = {
		bourbon_instability_variables = yes
		bourbon_instability_pulse_variables_progress = yes
		random_law = {
			limit = { type = law_type:law_monarchy }
			save_scope_as = monarchy_scope
		}
	}

	scripted_progress_bar = bourbon_instability_progress_bar
	display_progressbar_as_months = yes

	on_weekly_pulse = {
		events = {
		}
		effect = {
			if = {
				limit = {}
				bourbon_instability_pulse_variables_progress = yes
			}
		}
	}

	complete = {
		OR = {
			AND = {
				country_has_monarchy_law = yes
				custom_tooltip = {
					text = je_bourbon_instability_tt
					scope:journal_entry ?= {
						"scripted_bar_progress(bourbon_instability_progress_bar)" >= 100
					}
				}
			}
			AND = {
				country_has_monarchy_law = yes
				any_subject_or_below = {
					OR = {
						country_definition = cd:SCI
						was_formed_from = SCI
					}
				}
			}
		}
		has_technology_researched = nationalism
	}

	on_complete = {
		add_loyalists = {
			value = 0.35
			culture = cu:south_italian
		}
	}

	fail = {
		OR = {
			country_has_monarchy_law = no
			any_subject_or_below = {
				NOT = {
					country_definition = cd:SCI
					was_formed_from = SCI
				}
			}
		}
	}

	on_fail = {

			value = 0.15

		add_radicals = {

			culture = cu:siciliano
		}
	}

	should_be_pinned_by_default_uninvolved_or_context = yes
}
