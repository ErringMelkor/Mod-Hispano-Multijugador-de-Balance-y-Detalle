#curtail the aristocracy
je_age_of_nicholas_i_curtail_aristocracy = {
	name = "je_age_of_nicholas_i_curtail_aristocracy"
	desc = ""

	visible = {
		c:RUS = ROOT
	}

	possible = {
		OR = {
			has_law_or_variant = law_type:law_appointed_bureaucrats
			has_law_or_variant = law_type:law_elected_bureaucrats
		}
		authority >= 200
	}

	effect = {

		add_modifier = {
			name = modifier_curtailed_landowners
			years = 5
			is_decaying = yes
		}

	}
}

#appease the peasantry
je_age_of_nicholas_i_appease_peasantry = {
	name = "je_age_of_nicholas_i_appease_peasantry"
	desc = ""

	visible = {
		c:RUS = ROOT
	}

	possible = {
		OR = {
			country_turmoil >= 0.25
			radical_fraction = {
			value = 0.15
			pop_type = peasants
			}
		}
	}
	
	effect = {

		add_modifier = {
			name = modifier_appeased_peasantry
			years = 5
		}

		add_loyalists = {
			value = 0.1
			pop_type = peasants
		}

		add_loyalists = {
			value = 0.05
			pop_type = farmers
		}

	}
}
